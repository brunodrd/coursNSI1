\begin{algorithm}[H]
\DontPrintSemicolon

  \KwData{$n$ entier naturel}
  \KwResult{chaine correspondant à la conversion binaire de $n$}
  $resultat: chaine \gets ""$\;
  $quotient: entier \gets n$\;
  \If{$n=0$}{$resultat \gets "0"$}
  \Else{
  \While{$quotient \ne 0$}{
		$resultat \gets "$reste division quotient par 2$" + resultat$  \;
		$quotient \gets quotient//2$\;		
  		}
  }
  \Return {$resultat$}

\caption{Conversion binaire}
\end{algorithm}
